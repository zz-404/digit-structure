struct bigint {
    int arr[4500] = { 0 };
    int num = 0;
    bigint() {
        memset(arr, 0, sizeof(arr));
        num = 0;
    }
    bigint(long long x) {
        memset(arr, 0, sizeof(arr));
        num = 0;
        if (x == 0) {
            arr[0] = 0;
            num = 1;
            return;
        }
        while (x) {
            arr[num++] = x % 10;
            x /= 10;
        }
    }

    bigint operator*(const bigint& other) {
        bigint answer;
        for (int i = 0; i < num; i++) {
            for (int j = 0; j < other.num; j++) {
                answer.arr[i + j] += arr[i] * other.arr[j];
            }
        }
        answer.flatten(num + other.num);
        return answer;
    }

        return answer;

        num = x;

            num--;
        }
