struct bigint {
    int arr[4500] = { 0 };
    int num = 0;
    bigint() {
        memset(arr, 0, sizeof(arr));
        num = 0;
    }
    bigint(long long x) {
        memset(arr, 0, sizeof(arr));
        num = 0;
        if (x == 0) {
            arr[0] = 0;
            num = 1;
            return;
        }
        while (x) {
            arr[num++] = x % 10;
            x /= 10;
        }
    }

    bigint operator*(const bigint& other) {
        bigint answer;
        for (int i = 0; i < num; i++) {
            for (int j = 0; j < other.num; j++) {
                answer.arr[i + j] += arr[i] * other.arr[j];
            }
        }
        answer.flatten(num + other.num);
        return answer;
    }
    
    bigint operator/(const long long& x) {
        bigint answer;
        long long temp = 0;
        for (int i = num-1; i >=0; i--) {
            temp *= 10;
            temp += arr[i];
             
            answer.arr[i] = temp / x;
            temp %= x;     
            
        }
        
        answer.num = num;
        while (answer.num > 1 && answer.arr[answer.num - 1] == 0)
            answer.num--;
        
        return answer;
    }

    void flatten(int x) {
        num = x;
        for (int i = 0; i < num; i++) {
            arr[i + 1] += arr[i] / 10;
            arr[i] %= 10;
        }
        while (num>1 && arr[num-1] == 0) {
            num--;
        }
    }

    int cmp(const bigint& other) {
        if (num != other.num) {
            if (num > other.num)return 1;
            return -1;
        }
        else {
            for (int i = num - 1; i >= 0; i--) {
                if (arr[i] > other.arr[i]) {
                    return 1;
                }
                else if (arr[i] < other.arr[i]) {
                    return -1;
                }
            }
        }
        return 0;
    }
    int operator> (const bigint& other) {
        return cmp(other) == 1;
    }
    int operator== (const bigint& other) {
        return cmp(other) == 0;
    }
    int operator< (const bigint& other) {
        return cmp(other) == -1;

#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
#include <cassert>
using namespace std;

struct BigInt {
    vector<int> d; // 低位在前，高位在后
    // 去除前导零，同时保证至少有一位（数值0时保留一个0）
    void trim() {
        while (d.size() > 1 && d.back() == 0) d.pop_back();
    }

    // 构造函数
    BigInt() { d.push_back(0); }
    BigInt(long long x) {
        if (x == 0) d.push_back(0);
        else {
            while (x) {
                d.push_back(x % 10);
                x /= 10;
            }
        }
    }
    BigInt(const string& s) {
        for (int i = s.size() - 1; i >= 0; --i)
            d.push_back(s[i] - '0');
        trim();
    }

    // 输入输出
    friend istream& operator>>(istream& in, BigInt& a) {
        string s; in >> s;
        a.d.clear();
        for (int i = s.size() - 1; i >= 0; --i)
            a.d.push_back(s[i] - '0');
        a.trim();
        return in;
    }
    friend ostream& operator<<(ostream& out, const BigInt& a) {
        for (int i = a.d.size() - 1; i >= 0; --i)
            out << a.d[i];
        return out;
    }

    // 比较
    bool operator<(const BigInt& rhs) const {
        if (d.size() != rhs.d.size()) return d.size() < rhs.d.size();
        for (int i = d.size() - 1; i >= 0; --i)
            if (d[i] != rhs.d[i]) return d[i] < rhs.d[i];
        return false;
    }
    bool operator>(const BigInt& rhs) const { return rhs < *this; }
    bool operator<=(const BigInt& rhs) const { return !(rhs < *this); }
    bool operator>=(const BigInt& rhs) const { return !(*this < rhs); }
    bool operator==(const BigInt& rhs) const { return d == rhs.d; }
    bool operator!=(const BigInt& rhs) const { return !(*this == rhs); }

    // 加法
    BigInt operator+(const BigInt& rhs) const {
        BigInt res;
        res.d.clear();
        int carry = 0;
        for (size_t i = 0; i < max(d.size(), rhs.d.size()) || carry; ++i) {
            int x = carry;
            if (i < d.size()) x += d[i];
            if (i < rhs.d.size()) x += rhs.d[i];
            res.d.push_back(x % 10);
            carry = x / 10;
        }
        return res;
    }
    BigInt& operator+=(const BigInt& rhs) { *this = *this + rhs; return *this; }

    // 减法（要求 *this >= rhs，结果非负）
    BigInt operator-(const BigInt& rhs) const {
        assert(*this >= rhs);
        BigInt res;
        res.d.clear();
        int borrow = 0;
        for (size_t i = 0; i < d.size(); ++i) {
            int x = d[i] - borrow;
            if (i < rhs.d.size()) x -= rhs.d[i];
            borrow = 0;
            if (x < 0) { x += 10; borrow = 1; }
            res.d.push_back(x);
        }
        res.trim();
        return res;
    }
    BigInt& operator-=(const BigInt& rhs) { *this = *this - rhs; return *this; }

    // 乘法（高精度 × 高精度）
    BigInt operator*(const BigInt& rhs) const {
        BigInt res;
        res.d.assign(d.size() + rhs.d.size(), 0);
        for (size_t i = 0; i < d.size(); ++i)
            for (size_t j = 0; j < rhs.d.size(); ++j)
                res.d[i + j] += d[i] * rhs.d[j];
        for (size_t i = 0; i < res.d.size() - 1; ++i) {
            res.d[i + 1] += res.d[i] / 10;
            res.d[i] %= 10;
        }
        res.trim();
        return res;
    }
    BigInt& operator*=(const BigInt& rhs) { *this = *this * rhs; return *this; }

    // 乘法（高精度 × 低精度）
    BigInt operator*(int rhs) const {
        BigInt res;
        res.d.clear();
        long long carry = 0;
        for (size_t i = 0; i < d.size() || carry; ++i) {
            if (i < d.size()) carry += (long long)d[i] * rhs;
            res.d.push_back(carry % 10);
            carry /= 10;
        }
        res.trim();
        return res;
    }
    friend BigInt operator*(int lhs, const BigInt& rhs) { return rhs * lhs; }
    BigInt& operator*=(int rhs) { *this = *this * rhs; return *this; }

    // 除法与取余（高精度 ÷ 低精度），返回商，同时可传出余数
    BigInt dividedBy(int rhs, int& remainder) const {
        BigInt res;
        res.d.resize(d.size());
        long long r = 0;
        for (int i = d.size() - 1; i >= 0; --i) {
            r = r * 10 + d[i];
            res.d[i] = r / rhs;
            r %= rhs;
        }
        res.trim();
        remainder = r;
        return res;
    }
    BigInt operator/(int rhs) const {
        int rem;
        return dividedBy(rhs, rem);
    }
    int operator%(int rhs) const {
        int rem;
        dividedBy(rhs, rem);
        return rem;
    }

    // 除法（高精度 ÷ 高精度），采用试商减法
    BigInt operator/(const BigInt& rhs) const {
        assert(rhs != BigInt(0));
        BigInt dividend = *this, divisor = rhs, quotient;
        quotient.d.resize(dividend.d.size()); // 暂存
        // 商的位数
        int shift = dividend.d.size() - divisor.d.size();
        BigInt shiftedDivisor = divisor;
        if (shift > 0) {
            reverse(shiftedDivisor.d.begin(), shiftedDivisor.d.end());
            shiftedDivisor.d.resize(shiftedDivisor.d.size() + shift, 0);
            reverse(shiftedDivisor.d.begin(), shiftedDivisor.d.end());
        }
        while (shift >= 0) {
            while (dividend >= shiftedDivisor) {
                dividend -= shiftedDivisor;
                quotient.d[shift]++;
            }
            if (shift > 0) {
                // shiftedDivisor 右移一位（除以10）
                reverse(shiftedDivisor.d.begin(), shiftedDivisor.d.end());
                shiftedDivisor.d.pop_back();
                reverse(shiftedDivisor.d.begin(), shiftedDivisor.d.end());
            }
            --shift;
        }
        quotient.trim();
        return quotient;
    }
    BigInt operator%(const BigInt& rhs) const {
        return *this - (*this / rhs) * rhs;
    }
    BigInt& operator/=(const BigInt& rhs) { *this = *this / rhs; return *this; }
    BigInt& operator%=(const BigInt& rhs) { *this = *this % rhs; return *this; }
};

int main() {
    BigInt a, b;
    cin >> a >> b;

    cout << "a + b = " << a + b << endl;
    cout << "a - b = " << a - b << endl;   // 需要保证 a >= b
    cout << "a * b = " << a * b << endl;
    cout << "a / b = " << a / b << endl;
    cout << "a % b = " << a % b << endl;

    // 低精度运算
    cout << "a * 123 = " << a * 123 << endl;
    int remainder;
    BigInt q = a.dividedBy(7, remainder);
    cout << "a / 7 = " << q << ", a % 7 = " << remainder << endl;

    return 0;
}
