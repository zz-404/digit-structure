# 八皇后问题
- 以行号作为递归深度，否则如果在每个循环中都从一开始搜索行号的话会数重，例如把第一个皇后放2，4，第二个皇后放42和反转时完全没有区别(皇后之间没有区分)
- unorderedmap在值等于0时并不会自动移除键
- 在arr索引小于0时不一定非要用unorderedmap，可以发现其索引可以加定值n

## 深度搜索必须注意回溯，且深度搜索不适合求解最短路问题，宽度优先搜索利于求解最短路问题,且宽度优先搜索不需要回溯
## 深度搜索和宽度搜索都需要注意访问标记

int bfs() {
    queue<pair<int, int>> q; // (楼层, 步数)
    q.push({a, 0});
    vis[a] = true;
    while (!q.empty()) {
        auto [cur, step] = q.front(); q.pop();
        if (cur == b) return step;
        int nxt = cur + arr[cur];
        if (nxt >= 1 && nxt <= n && !vis[nxt]) {
            vis[nxt] = true;//注意，在准备访问时就把要访问的设置visit，否则可能导致memory limit
            q.push({nxt, step + 1});
        }
        nxt = cur - arr[cur];
        if (nxt >= 1 && nxt <= n && !vis[nxt]) {
            vis[nxt] = true;
            q.push({nxt, step + 1});
        }
    }
    return -1;
}

queue<pair<int, int>> q;
    q.push({0, 0});
    vis[0][0] = true;
    int step = 0;  // 当前步数（即当前队列中节点的时间）

            q.pop();

            for (int d = 0; d < 4; ++d) {
                int nx = x + dx[d];
                int ny = y + dy[d];

                // 不能走到负坐标
                if (nx < 0 || ny < 0) continue;

                // 如果走出地图边界，认为找到了安全格子
                if (nx >= MAX || ny >= MAX) {
                    cout << step + 1 << '\n';
                    return 0;
                }

                if (vis[nx][ny]) continue;

                // 下一步时间 = step + 1，必须严格小于该格被摧毁时间
                if (danger[nx][ny] <= step + 1) continue;

                // 如果该格永远不会被摧毁，直接返回
                if (danger[nx][ny] == INF) {
                    cout << step + 1 << '\n';
                    return 0;
                }

                vis[nx][ny] = true;
                q.push({nx, ny});
            }
        }
        step++;
    }

    cout << -1 << '\n';
    return 0;
}
